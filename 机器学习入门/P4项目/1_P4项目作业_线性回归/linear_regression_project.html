def matxMultiply(A, B):
    shape_A = shape(A);
    shape_B = shape(B);
    
    if(shape_A[1]!=shape_B[0]):
        raise ValueError;
    elif(shape_A[0]*shape_A[1]*shape_B[0]*shape_B[1]==0):
        raise ValueError;
    else:
        pass;
    
    A_B_product = [];
    for row in range(shape_A[0]):#A
        new_row = [];
        for col in range(shape_B[1]):#B
            B_col = [x[col] for x in B];#
            A_B_dot = sum([x*y for x,y in zip(A[row],B_col)]);
            new_row.append(A_B_dot);            
        A_B_product.append(new_row);
        
    return A_B_product
